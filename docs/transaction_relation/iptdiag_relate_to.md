SELECT
    d.an,
    d.hn,
    d.diagtype,
    dt.name AS diagtype_name,
    d.icd10,
    icd.name AS icd10_name,
    d.doctor,
    doc.name AS doctor_name
FROM iptdiag d
LEFT JOIN diagtype dt ON dt.diagtype = d.diagtype
LEFT JOIN icd101 icd ON icd.code = d.icd10
LEFT JOIN doctor doc ON doc.code = d.doctor
WHERE d.an = '<AN>';
