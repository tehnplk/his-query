## Common Query Pattern

SELECT
    od.vn,
    od.hn,
    od.vstdate,
    od.diagtype,
    dt.name AS diagtype_name,
    od.icd10,
    icd.name AS icd10_name,
    od.doctor,
    d.name AS doctor_name
FROM ovstdiag od
LEFT JOIN diagtype dt ON dt.diagtype = od.diagtype
LEFT JOIN icd101 icd ON icd.code = od.icd10
LEFT JOIN doctor d ON d.code = od.doctor
WHERE od.vn = '<VN>';
