## Common Query Pattern
```sql
SELECT
    v.vn,
    v.hn,
    p.cid,
    CONCAT(p.pname, p.fname, ' ', p.lname) AS patient_name,
    v.vstdate,
    v.pttype,
    pt.name AS pttype_name,
    v.pdx,
    icd.name AS pdx_name,
    v.dx_doctor,
    d.name AS dx_doctor_name,
    v.income,
    v.paid_money,
    v.remain_money
FROM vn_stat v
LEFT JOIN patient p ON p.hn = v.hn
LEFT JOIN ovst o ON o.vn = v.vn
LEFT JOIN pttype pt ON pt.pttype = v.pttype
LEFT JOIN icd101 icd ON icd.code = v.pdx
LEFT JOIN doctor d ON d.code = v.dx_doctor
WHERE v.vstdate BETWEEN '2026-02-01' AND '2026-02-29';
```
